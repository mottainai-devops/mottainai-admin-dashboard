import 'dart:convert';
import 'package:http/http.dart' as http;

class LotService {
  static const String apiUrl = 'https://admin.kowope.xyz/api/trpc/lots.list';
  
  Future<List<Lot>> fetchLots() async {
    final response = await http.get(Uri.parse(apiUrl));
    
    if (response.statusCode == 200) {
      final data = json.decode(response.body);
      final lots = data['result']['data']['json'] as List;
      return lots.map((lot) => Lot.fromJson(lot)).toList();
    } else {
      throw Exception('Failed to load lots');
    }
  }
}

class Lot {
  final int objectId;
  final int lotId;
  final String wardName;
  final String businessName;
  
  Lot({
    required this.objectId,
    required this.lotId,
    required this.wardName,
    required this.businessName,
  });
  
  factory Lot.fromJson(Map<String, dynamic> json) {
    return Lot(
      objectId: json['OBJECTID'],
      lotId: json['Lot_ID'],
      wardName: json['ward_name'] ?? '',
      businessName: json['Business_Name'] ?? '',
    );
  }
}
